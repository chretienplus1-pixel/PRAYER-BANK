            {/* Sub-theme selector */}
            <div style={{ display:"flex", gap:4, flexWrap:"wrap", marginBottom:10 }}>
              {curTheme?.sub.map((s,i) => (
                <button key={i} onClick={() => setThemeSubIdx(i)} style={{ background:themeSubIdx===i?`linear-gradient(135deg, ${curTheme.color}33, ${curTheme.color}11)`:"transparent", color:themeSubIdx===i?curTheme.color:GOLD, border:`1px solid ${themeSubIdx===i?curTheme.color:GOLD_DIM}`, borderRadius:10, padding:"5px 10px", fontSize:9, cursor:"pointer", fontFamily:"Georgia,serif", letterSpacing:0.5 }}>
                  {s.label}
                </button>
              ))}
            </div>

            {/* Sub-theme description */}
            {curSub?.description && (
              <div style={{ background:"rgba(201,168,76,0.05)", border:`1px solid ${GOLD_FAINT}`, borderLeft:`2px solid ${curTheme?.color||GOLD}`, borderRadius:8, padding:"8px 12px", marginBottom:10 }}>
                <p style={{ color:"#c8b890", fontSize:12, lineHeight:1.65, margin:0, fontFamily:"'EB Garamond',Georgia,serif", fontStyle:"italic" }}>{curSub.description}</p>
              </div>
            )}

            {/* Favorites toggle */}
            <div style={{ display:"flex", gap:6, marginBottom:10, alignItems:"center" }}>
              <Btn small active={showFavorites} onClick={() => setShowFavorites(f=>!f)}>⭐ {t.favorites} ({favCount})</Btn>
              <div style={{ flex:1 }} />
              <div style={{ display:"flex", alignItems:"center", gap:6 }}>
                <span style={{ color:"#6b5c3e", fontSize:9, fontFamily:"Georgia,serif" }}>Auto</span>
                <button onClick={() => setAutoRead(a=>!a)} style={{ background:autoRead?GOLD_GLOW:GOLD_FAINT, border:`1px solid ${autoRead?GOLD:GOLD_DIM}`, borderRadius:12, padding:"3px 8px", color:autoRead?GOLD:"#6b5c3e", fontSize:9, cursor:"pointer", fontFamily:"Georgia,serif" }}>🔊</button>
              </div>
            </div>

            {/* Prayer list */}
            {showFavorites ? (
              Object.keys(favorites).length === 0 ? <p style={{ color:"#6b5c3e", textAlign:"center", padding:"20px 0", fontSize:13 }}>{t.noFav}</p> :
              Object.entries(favorites).map(([key, item]) => (
                <PrayerCard key={key} text={item.text} index={0} checked={!!checked[key]} onCheck={() => toggleChecked(key)} onSpeak={() => speak(item.text, key)} onFav={() => toggleFav(key, item)} onCopy={() => copyText(key, item.text)} isFav={true} isCopied={copiedId===key} ttsActive={activeTtsPrayerKey===key} textColor={textColor} themeColor={GOLD} />
              ))
            ) : (
              curSub?.prayers.map((p, i) => {
                const key = `${themeIdx}-${themeSubIdx}-${i}`;
                return <PrayerCard key={key} text={p} index={i} checked={!!checked[key]} onCheck={() => toggleChecked(key)} onSpeak={() => speak(p, key)} onFav={() => toggleFav(key, {text:p, label:curSub.label})} onCopy={() => copyText(key, p)} isFav={!!favorites[key]} isCopied={copiedId===key} ttsActive={activeTtsPrayerKey===key} textColor={textColor} themeColor={curTheme?.color||GOLD} />;
              })
            )}

            {/* Progress */}
            <Card style={{ marginTop:6 }}>
              <Label>{t.progress}</Label>
              <ProgressBar value={completedCount} max={totalPrayers} />
              <p style={{ color:"#8b6914", fontSize:10, margin:"6px 0 0", fontFamily:"Georgia,serif", textAlign:"right" }}>{completedCount}/{totalPrayers} prières accomplies</p>
            </Card>

            {/* Timer */}
            <Card>
              <Label accent>{t.timer}</Label>
              <div style={{ display:"flex", alignItems:"center", gap:8, marginBottom:10 }}>
                <div style={{ fontSize:28, fontFamily:"'Cinzel',Georgia,serif", color:GOLD, letterSpacing:2, flex:1, textAlign:"center" }}>{fmtSecs(running||seconds>0?seconds:timerMin*60)}</div>
                <input type="range" min={5} max={120} step={5} value={timerMin} onChange={e => { setTimerMin(Number(e.target.value)); setSeconds(0); }} style={{ width:80, accentColor:GOLD }} />
                <span style={{ color:"#8b6914", fontSize:10, fontFamily:"Georgia,serif", minWidth:30 }}>{timerMin}m</span>
              </div>
              <div style={{ display:"flex", gap:6 }}>
                <Btn active={running} onClick={() => { if(!running && seconds===0) setSeconds(timerMin*60); setRunning(r=>!r); }}>{running?t.pause:t.start}</Btn>
                <Btn onClick={() => { setRunning(false); setSeconds(0); }}>{t.reset}</Btn>
              </div>
            </Card>

            {/* Prayer time tracking */}
            <Card>
              <Label accent>⏱ SUIVI DU TEMPS DE PRIÈRE</Label>
              <div style={{ display:"flex", alignItems:"center", gap:10, marginBottom:8 }}>
                <div style={{ flex:1 }}>
                  <p style={{ color:"#8b6914", fontSize:9, margin:"0 0 4px", fontFamily:"Georgia,serif" }}>Prière: {prayerTimeDone}/{prayerTimeGoal} min</p>
                  <ProgressBar value={prayerTimeDone} max={prayerTimeGoal} color={GOLD} />
                </div>
                <button onClick={() => { const n=prayerTimeDone+5; setPrayerTimeDone(n); persist("pb-prayer-time-done",n); }} style={{ background:GOLD_FAINT, border:`1px solid ${GOLD_DIM}`, borderRadius:8, padding:"4px 8px", color:GOLD, fontSize:10, cursor:"pointer" }}>+5m</button>
              </div>
              <div style={{ display:"flex", alignItems:"center", gap:10 }}>
                <div style={{ flex:1 }}>
                  <p style={{ color:"#8b6914", fontSize:9, margin:"0 0 4px", fontFamily:"Georgia,serif" }}>Lecture Bible: {bibleTimeDone}/{bibleTimeGoal} min</p>
                  <ProgressBar value={bibleTimeDone} max={bibleTimeGoal} color="#4fc3f7" />
                </div>
                <button onClick={() => { const n=bibleTimeDone+5; setBibleTimeDone(n); persist("pb-bible-time-done",n); }} style={{ background:"rgba(79,195,247,0.08)", border:`1px solid rgba(79,195,247,0.25)`, borderRadius:8, padding:"4px 8px", color:"#4fc3f7", fontSize:10, cursor:"pointer" }}>+5m</button>
              </div>
            </Card>
          </div>
        )}

        {/* ─── TAB 1: Session Guidée ─── */}
        {tab===1 && (
          <div className="fade-in">
            {!sessRunning && !sessDone && sessionPrayers.length===0 && (
              <Card>
                <Label accent>{t.session}</Label>
                <div style={{ marginBottom:10 }}>
                  <p style={{ color:"#8b6914", fontSize:9, margin:"0 0 4px", fontFamily:"Georgia,serif", letterSpacing:1 }}>THÈME</p>
                  <select value={sessTheme} onChange={e => setSessTheme(e.target.value)} style={{ width:"100%", background:"rgba(201,168,76,0.06)", border:`1px solid ${GOLD_DIM}`, borderRadius:8, padding:"7px 10px", color:textColor, fontSize:12, fontFamily:"Georgia,serif" }}>
                    <option value="all">Tous les thèmes</option>
                    {THEMES_DATA.map(th => <option key={th.id} value={th.id}>{th.icon} {th.label}</option>)}
                  </select>
                </div>
                <div style={{ display:"flex", gap:12, marginBottom:16 }}>
                  <div style={{ flex:1 }}>
                    <p style={{ color:"#8b6914", fontSize:9, margin:"0 0 4px", fontFamily:"Georgia,serif", letterSpacing:1 }}>{t.duration} (min)</p>
                    <input type="number" min={5} max={120} value={sessDuration} onChange={e => setSessDuration(Number(e.target.value))} style={{ width:"100%", background:"rgba(201,168,76,0.06)", border:`1px solid ${GOLD_DIM}`, borderRadius:8, padding:"7px 10px", color:textColor, fontSize:13, fontFamily:"Georgia,serif", boxSizing:"border-box" }} />
                  </div>
                  <div style={{ flex:1 }}>
                    <p style={{ color:"#8b6914", fontSize:9, margin:"0 0 4px", fontFamily:"Georgia,serif", letterSpacing:1 }}>{t.interval}</p>
                    <input type="number" min={1} max={15} value={sessInterval} onChange={e => setSessInterval(Number(e.target.value))} style={{ width:"100%", background:"rgba(201,168,76,0.06)", border:`1px solid ${GOLD_DIM}`, borderRadius:8, padding:"7px 10px", color:textColor, fontSize:13, fontFamily:"Georgia,serif", boxSizing:"border-box" }} />
                  </div>
                </div>
                <button onClick={buildSession} style={{ width:"100%", background:`linear-gradient(135deg, ${GOLD}, #a87a2a)`, border:"none", borderRadius:12, padding:"12px", color:"#1a0f00", fontSize:13, fontWeight:"bold", cursor:"pointer", fontFamily:"'Cinzel',Georgia,serif", letterSpacing:2 }}>{t.build}</button>
              </Card>
            )}

            {sessionPrayers.length > 0 && !sessDone && (
              <div>
                <Card glow>
                  <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center", marginBottom:12 }}>
                    <Label>{sessIdx+1}/{sessionPrayers.length}</Label>
                    <div style={{ fontSize:22, fontFamily:"'Cinzel',Georgia,serif", color:GOLD }}>{fmtSecs(sessSecsLeft)}</div>
                  </div>
                  <ProgressBar value={sessIdx} max={sessionPrayers.length-1} style={{ marginBottom:12 }} />
                  <p style={{ color:"#8b6914", fontSize:9, letterSpacing:1, margin:"0 0 8px", fontFamily:"'Cinzel',Georgia,serif" }}>{sessionPrayers[sessIdx]?.label}</p>
                  <p style={{ color:textColor, fontSize:15, lineHeight:1.8, margin:"0 0 12px", fontStyle:"italic", fontFamily:"'EB Garamond',Georgia,serif" }}>{sessionPrayers[sessIdx]?.text}</p>
                  <div style={{ display:"flex", gap:8, flexWrap:"wrap" }}>
                    <Btn active={sessRunning} onClick={() => setSessRunning(r=>!r)}>{sessRunning?"⏸ Pause":"▶ Démarrer"}</Btn>
                    {sessIdx > 0 && <Btn onClick={() => { setSessIdx(i=>i-1); setSessSecsLeft(sessInterval*60); }}>{t.prev}</Btn>}
                    {sessIdx < sessionPrayers.length-1 && <Btn onClick={() => { setSessIdx(i=>i+1); setSessSecsLeft(sessInterval*60); }}>{t.next}</Btn>}
                    <Btn onClick={() => speak(sessionPrayers[sessIdx]?.text||"")}>{t.read}</Btn>
                    <Btn onClick={() => { setSessRunning(false); setSessionPrayers([]); setSessDone(false); }}>↺ Reset</Btn>
                  </div>
                </Card>
                <p style={{ color:"#6b5c3e", fontSize:10, textAlign:"center", fontFamily:"Georgia,serif", marginTop:8 }}>Session : {sessDuration} min · {sessInterval} min par prière · {sessionPrayers.length} requêtes</p>
              </div>
            )}

            {sessDone && (
              <Card glow style={{ textAlign:"center", padding:"30px 20px" }}>
                <div style={{ fontSize:48, marginBottom:12 }}>🙏</div>
                <p style={{ color:GOLD, fontSize:16, fontFamily:"'Cinzel',Georgia,serif", letterSpacing:2, margin:"0 0 8px" }}>{t.sessionDone}</p>
                <p style={{ color:"#8b6914", fontSize:12, margin:"0 0 16px" }}>Sessions complétées : {sessCount}</p>
                <Btn onClick={() => { setSessionPrayers([]); setSessDone(false); }}>↺ Nouvelle session</Btn>
              </Card>
            )}

            {/* Prayer times alarm */}
            <Card style={{ marginTop:10 }}>
              <Label accent>⏰ RAPPELS DE PRIÈRE</Label>
              <div style={{ display:"flex", gap:6, flexWrap:"wrap", marginBottom:8 }}>
                {prayerTimes.map(pt => (
                  <div key={pt} style={{ background:GOLD_FAINT, border:`1px solid ${GOLD_DIM}`, borderRadius:8, padding:"4px 10px", display:"flex", alignItems:"center", gap:4 }}>
                    <span style={{ color:GOLD, fontSize:11, fontFamily:"Georgia,serif" }}>{pt}</span>
                    <button onClick={() => { const n=prayerTimes.filter(x=>x!==pt); setPrayerTimes(n); persist("pb-times",n); }} style={{ background:"transparent", border:"none", color:"#6b5c3e", cursor:"pointer", fontSize:12, padding:0, lineHeight:1 }}>×</button>
                  </div>
                ))}
              </div>
              <div style={{ display:"flex", gap:6 }}>
                <input type="time" value={newTime} onChange={e => setNewTime(e.target.value)} style={{ background:"rgba(201,168,76,0.06)", border:`1px solid ${GOLD_DIM}`, borderRadius:8, padding:"5px 8px", color:textColor, fontSize:12, fontFamily:"Georgia,serif", flex:1 }} />
                <Btn small onClick={() => { if(!prayerTimes.includes(newTime)) { const n=[...prayerTimes,newTime].sort(); setPrayerTimes(n); persist("pb-times",n); } }}>{t.addTime}</Btn>
              </div>
            </Card>
          </div>
        )}

        {/* ─── TAB 2: Actes Prophétiques ─── */}
        {tab===2 && (
          <div className="fade-in">
            <Card style={{ marginBottom:10 }}>
              <Label accent>✋ ACTES PROPHÉTIQUES</Label>
              <p style={{ color:"#c8b890", fontSize:12, lineHeight:1.6, margin:0, fontFamily:"'EB Garamond',Georgia,serif", fontStyle:"italic" }}>
                Les actes prophétiques sont des actions physiques ordonnées par Dieu qui créent une réalité spirituelle. Ils activent la foi, ouvrent des brèches et libèrent la puissance divine dans des situations bloquées.
              </p>
            </Card>
            <div style={{ display:"flex", gap:4, flexWrap:"wrap", marginBottom:10 }}>
              {actCats.map(c => <Btn key={c} small active={actCatFilter===c} onClick={() => setActCatFilter(c)}>{c}</Btn>)}
            </div>
            {ACTS.filter(a => actCatFilter==="Tous" || a.cat===actCatFilter).map((a,i) => (
              <Card key={i} style={{ cursor:"pointer", borderLeft:`3px solid ${GOLD_DIM}`, borderColor:expandedAct===i?GOLD:GOLD_DIM }} onClick={() => setExpandedAct(expandedAct===i?null:i)}>
                <div style={{ display:"flex", alignItems:"center", gap:10 }}>
                  <span style={{ fontSize:28, flexShrink:0 }}>{a.icon}</span>
                  <div style={{ flex:1 }}>
                    <p style={{ color:GOLD, fontSize:13, fontWeight:"bold", margin:"0 0 2px", fontFamily:"'Cinzel',Georgia,serif" }}>{a.title}</p>
                    <p style={{ color:"#8b6914", fontSize:9, margin:0, fontFamily:"Georgia,serif" }}>{a.cat} · {a.combat} · {a.ref}</p>
                  </div>
                  <span style={{ color:GOLD, fontSize:12 }}>{expandedAct===i?"▲":"▼"}</span>
                </div>
                {expandedAct===i && (
                  <div className="fade-in" style={{ marginTop:12 }} onClick={e => e.stopPropagation()}>
                    <div style={{ background:"rgba(40,30,10,0.5)", borderRadius:10, padding:"10px 12px", marginBottom:10 }}>
                      <p style={{ color:"#8b6914", fontSize:8, letterSpacing:2, margin:"0 0 6px" }}>INSTRUCTIONS</p>
                      <p style={{ color:textColor, fontSize:13, lineHeight:1.7, margin:0, fontFamily:"'EB Garamond',Georgia,serif" }}>{a.instruction}</p>
                    </div>
                    <div style={{ display:"flex", gap:8 }}>
                      <button onClick={() => speak(a.instruction)} style={{ background:"transparent", border:"none", color:GOLD, cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>▶ Lire l'instruction</button>
                      {ttsPlaying && <button onClick={stopSpeak} style={{ background:"transparent", border:"none", color:"#e87878", cursor:"pointer", fontSize:11 }}>⏹ Stop</button>}
                    </div>
                  </div>
                )}
              </Card>
            ))}
          </div>
        )}

        {/* ─── TAB 3: Déclarations Prophétiques ─── */}
        {tab===3 && (
          <div className="fade-in">
            <Card style={{ marginBottom:10 }}>
              <Label accent>📣 DÉCLARATIONS PROPHÉTIQUES</Label>
              <p style={{ color:"#c8b890", fontSize:12, lineHeight:1.6, margin:0, fontFamily:"'EB Garamond',Georgia,serif", fontStyle:"italic" }}>
                La parole confessée avec foi crée la réalité. Romains 10:10 : « C'est par la bouche qu'on fait la confession qui sauve. » Proclamez ces déclarations à haute voix avec conviction.
              </p>
            </Card>
            <div style={{ display:"flex", gap:4, flexWrap:"wrap", marginBottom:10 }}>
              {declCats.map(c => <Btn key={c} small active={declCatFilter===c} onClick={() => setDeclCatFilter(c)}>{c}</Btn>)}
            </div>
            {PROPHETIC.filter(p => declCatFilter==="Tous" || p.cat===declCatFilter).map((p,i) => (
              <Card key={i} style={{ borderLeft:`3px solid ${GOLD_DIM}` }}>
                <div style={{ display:"flex", gap:10, alignItems:"flex-start" }}>
                  <div style={{ flex:1 }}>
                    <p style={{ color:"#8b6914", fontSize:8, letterSpacing:1, margin:"0 0 5px", fontFamily:"'Cinzel',Georgia,serif" }}>{p.cat} · {p.ref}</p>
                    <p style={{ color:textColor, fontSize:15, lineHeight:1.8, fontStyle:"italic", margin:"0 0 10px", fontFamily:"'EB Garamond',Georgia,serif" }}>« {p.text} »</p>
                    <div style={{ display:"flex", gap:8 }}>
                      <button onClick={() => { speak(p.text); updateChallenge("check"); }} style={{ background:"transparent", border:"none", color:GOLD, cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>▶ Proclamer</button>
                      <button onClick={() => copyText(`decl-${i}`, p.text)} style={{ background:"transparent", border:"none", color:copiedId===`decl-${i}`?GOLD:"#6b5c3e", cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>
                        {copiedId===`decl-${i}`?"✓ Copié":"📋"}
                      </button>
                      <button onClick={() => toggleFav(`decl-${i}`, {text:p.text, label:p.cat})} style={{ background:"transparent", border:"none", color:favorites[`decl-${i}`]?GOLD:"#6b5c3e", cursor:"pointer", fontSize:14 }}>
                        {favorites[`decl-${i}`]?"★":"☆"}
                      </button>
                    </div>
                  </div>
                </div>
              </Card>
            ))}
          </div>
        )}

        {/* ─── TAB 4: Esprits à Combattre ─── */}
        {tab===4 && (
          <div className="fade-in">
            <Card style={{ marginBottom:10 }}>
              <Label accent>👁 ESPRITS À COMBATTRE</Label>
              <p style={{ color:"#c8b890", fontSize:12, lineHeight:1.6, margin:0, fontFamily:"'EB Garamond',Georgia,serif", fontStyle:"italic" }}>
                Éphésiens 6:12 : « Nous n'avons pas à lutter contre la chair et le sang, mais contre les dominations, contre les autorités, contre les princes des ténèbres. » Identifier l'ennemi est la première étape de la victoire.
              </p>
            </Card>
            <div style={{ display:"flex", gap:4, flexWrap:"wrap", marginBottom:10 }}>
              {spiritCats.map(c => <Btn key={c} small active={spiritCatFilter===c} onClick={() => setSpiritCatFilter(c)}>{c}</Btn>)}
            </div>
            {SPIRITS.filter(s => spiritCatFilter==="Tous" || s.cat===spiritCatFilter).map((s,i) => (
              <Card key={i} style={{ cursor:"pointer", borderLeft:`3px solid ${CRIMSON}44`, borderColor:expandedSpirit===i?"#ef5350":GOLD_DIM }} onClick={() => setExpandedSpirit(expandedSpirit===i?null:i)}>
                <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center" }}>
                  <div>
                    <p style={{ color:"#ef5350", fontSize:13, fontWeight:"bold", margin:"0 0 2px", fontFamily:"'Cinzel',Georgia,serif" }}>{s.name}</p>
                    <p style={{ color:"#8b6914", fontSize:9, margin:0, fontFamily:"Georgia,serif" }}>{s.cat} · {s.ref}</p>
                  </div>
                  <span style={{ color:GOLD, fontSize:12 }}>{expandedSpirit===i?"▲":"▼"}</span>
                </div>
                {expandedSpirit===i && (
                  <div className="fade-in" style={{ marginTop:10 }} onClick={e => e.stopPropagation()}>
                    <div style={{ background:"rgba(180,30,20,0.1)", border:"1px solid rgba(180,30,20,0.2)", borderRadius:8, padding:"8px 10px", marginBottom:10 }}>
                      <p style={{ color:"#e87878", fontSize:8, letterSpacing:1, margin:"0 0 4px" }}>DESCRIPTION</p>
                      <p style={{ color:textColor, fontSize:12, lineHeight:1.6, margin:0, fontFamily:"'EB Garamond',Georgia,serif" }}>{s.description}</p>
                    </div>
                    <div style={{ background:"rgba(40,30,10,0.5)", borderRadius:8, padding:"8px 10px" }}>
                      <p style={{ color:"#8b6914", fontSize:8, letterSpacing:1, margin:"0 0 4px" }}>PRIÈRE DE COMBAT</p>
                      <p style={{ color:textColor, fontSize:13, lineHeight:1.7, fontStyle:"italic", margin:"0 0 8px", fontFamily:"'EB Garamond',Georgia,serif" }}>{s.prayer}</p>
                      <div style={{ display:"flex", gap:8 }}>
                        <button onClick={() => speak(s.prayer)} style={{ background:"transparent", border:"none", color:GOLD, cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>▶ Proclamer</button>
                        <button onClick={() => toggleFav(`spirit-${i}`, {text:s.prayer, label:s.name})} style={{ background:"transparent", border:"none", color:favorites[`spirit-${i}`]?GOLD:"#6b5c3e", cursor:"pointer", fontSize:14 }}>{favorites[`spirit-${i}`]?"★":"☆"}</button>
                      </div>
                    </div>
                  </div>
                )}
              </Card>
            ))}
          </div>
        )}

        {/* ─── TAB 5: Bible Intégrale ─── */}
        {tab===5 && (
          <div className="fade-in">
            <Card>
              <Label accent>📖 BIBLE INTÉGRALE</Label>
              <div style={{ display:"flex", gap:6, marginBottom:10 }}>
                {["AT","NT"].map(s => <Btn key={s} small active={bibleSection===s} onClick={() => { setBibleSection(s); setBibleBook(null); setBibleChapter(null); }}>{s==="AT"?"Ancien Testament":"Nouveau Testament"}</Btn>)}
              </div>
              {!bibleBook && (
                <div style={{ display:"flex", gap:6, flexWrap:"wrap" }}>
                  {BIBLE_BOOKS[bibleSection].map(book => (
                    <button key={book.abbr} onClick={() => { setBibleBook(book); setBibleChapter(null); }} style={{ background:GOLD_FAINT, border:`1px solid ${GOLD_DIM}`, borderRadius:8, padding:"5px 9px", color:GOLD, fontSize:10, cursor:"pointer", fontFamily:"Georgia,serif" }}>{book.name}</button>
                  ))}
                </div>
              )}
              {bibleBook && !bibleChapter && (
                <div>
                  <div style={{ display:"flex", alignItems:"center", gap:8, marginBottom:10 }}>
                    <button onClick={() => setBibleBook(null)} style={{ background:"transparent", border:"none", color:"#8b6914", cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>← Retour</button>
                    <p style={{ color:GOLD, fontSize:13, fontWeight:"bold", margin:0, fontFamily:"'Cinzel',Georgia,serif" }}>{bibleBook.name}</p>
                  </div>
                  <div style={{ display:"flex", gap:5, flexWrap:"wrap" }}>
                    {Array.from({length:bibleBook.chapters},(_,i)=>i+1).map(ch => (
                      <button key={ch} onClick={() => setBibleChapter(ch)} style={{ background:GOLD_FAINT, border:`1px solid ${GOLD_DIM}`, borderRadius:8, padding:"6px 10px", color:GOLD, fontSize:11, cursor:"pointer", fontFamily:"Georgia,serif", minWidth:36, textAlign:"center" }}>{ch}</button>
                    ))}
                  </div>
                </div>
              )}
              {bibleBook && bibleChapter && (
                <div>
                  <div style={{ display:"flex", alignItems:"center", gap:8, marginBottom:12 }}>
                    <button onClick={() => setBibleChapter(null)} style={{ background:"transparent", border:"none", color:"#8b6914", cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>← Chapitres</button>
                    <p style={{ color:GOLD, fontSize:14, fontWeight:"bold", margin:0, fontFamily:"'Cinzel',Georgia,serif" }}>{bibleBook.name} {bibleChapter}</p>
                  </div>
                  <div style={{ background:"rgba(201,168,76,0.05)", border:`1px solid ${GOLD_FAINT}`, borderRadius:10, padding:"12px", textAlign:"center" }}>
                    <p style={{ color:"#8b6914", fontSize:11, fontStyle:"italic", lineHeight:1.7, margin:0, fontFamily:"'EB Garamond',Georgia,serif" }}>Lecture de {bibleBook.name} chapitre {bibleChapter}</p>
                    <a href={`https://www.biblegateway.com/passage/?search=${bibleBook.abbr}+${bibleChapter}&version=LSG`} target="_blank" rel="noopener noreferrer" style={{ display:"inline-block", marginTop:12, background:`linear-gradient(135deg, ${GOLD}, #a87a2a)`, color:"#1a0f00", border:"none", borderRadius:10, padding:"8px 18px", fontSize:11, fontFamily:"'Cinzel',Georgia,serif", textDecoration:"none", letterSpacing:1 }}>

                    </a>

                </div>
              )}
            </Card>
            {/* Verse scroll banner */}
            <Card style={{ cursor:"pointer" }} onClick={() => setBibleVerseScroll(v=>!v)}>
              <Label accent>{t.scrollVerses}</Label>
              <p style={{ color:textColor, fontSize:14, fontStyle:"italic", lineHeight:1.7, margin:"0 0 6px", fontFamily:"'EB Garamond',Georgia,serif" }}>« {PRAYER_VERSES[verseScrollIdx].text} »</p>
              <p style={{ color:"#8b6914", fontSize:10, margin:0, fontFamily:"Georgia,serif" }}>— {PRAYER_VERSES[verseScrollIdx].ref} · {bibleVerseScroll?"Auto-défilement actif":"Taper pour activer"}</p>
            </Card>
          </div>
        )}

        {/* ─── TAB 6: Journal ─── */}
        {tab===6 && (
          <div className="fade-in">
            <Card>
              <Label accent>{t.journal}</Label>
              <div style={{ display:"flex", gap:6, marginBottom:8 }}>
                {["🙏","🔥","💪","😢","✨","🌅"].map(m => (
                  <button key={m} onClick={() => setJMood(m)} style={{ background:jMood===m?GOLD_GLOW:"transparent", border:`1px solid ${jMood===m?GOLD:GOLD_DIM}`, borderRadius:8, width:32, height:32, cursor:"pointer", fontSize:16 }}>{m}</button>
                ))}
              </div>
              <div style={{ display:"flex", gap:6, marginBottom:8 }}>
                {["Prière","Intercession","Louange","Délivrance","Action de grâce"].map(cat => (

              </div>
              <textarea value={jText} onChange={e => setJText(e.target.value)} placeholder={t.entry} rows={4} style={{ width:"100%", background:"rgba(201,168,76,0.04)", border:`1px solid ${GOLD_DIM}`, borderRadius:10, padding:"10px", color:textColor, fontSize:13, fontFamily:"'EB Garamond',Georgia,serif", resize:"vertical", outline:"none", lineHeight:1.7, boxSizing:"border-box" }} />

            </Card>

              <Card key={e.id} style={{ borderLeft:`2px solid ${GOLD_DIM}` }}>

                  <div style={{ display:"flex", gap:6, alignItems:"center" }}>
                    <span style={{ fontSize:16 }}>{e.mood}</span>
                    <div>
                      <p style={{ color:GOLD, fontSize:9, margin:0, fontFamily:"'Cinzel',Georgia,serif", letterSpacing:1 }}>{e.category}</p>

                  </div>

                </div>
                <p style={{ color:textColor, fontSize:13, lineHeight:1.7, margin:0, fontFamily:"'EB Garamond',Georgia,serif" }}>{e.text}</p>
              </Card>
            ))}
            {journal.length===0 && <p style={{ color:"#6b5c3e", textAlign:"center", padding:"20px 0", fontSize:12 }}>Aucune entrée dans le journal.</p>}
          </div>

              <div style={{ display:"grid", gridTemplateColumns:"1fr 1fr", gap:10 }}>

                  { label:t.totalPrayers, value:totalPrayers, icon:"🙏" },
                  { label:t.completedPrayers, value:completedCount, icon:"✓" },

              <Label accent>⏱ TEMPS CONSACRÉ AUJOURD'HUI</Label>
              <div style={{ marginBottom:10 }}>
                <div style={{ display:"flex", justifyContent:"space-between", marginBottom:4 }}>
                  <span style={{ color:"#8b6914", fontSize:11, fontFamily:"Georgia,serif" }}>🙏 Prière</span>
                  <span style={{ color:GOLD, fontSize:11, fontFamily:"Georgia,serif" }}>{prayerTimeDone}/{prayerTimeGoal} min</span>

                <div style={{ display:"flex", justifyContent:"space-between", marginBottom:4 }}>
                  <span style={{ color:"#8b6914", fontSize:11, fontFamily:"Georgia,serif" }}>📖 Lecture</span>

              <Label accent>{t.dailyChallenges}</Label>

            </Card>
            <Card>
              <Label accent>{t.weeklyChallenges}</Label>
              {challenges.weekly.map((c,i) => (
                <div key={i} style={{ marginBottom:12 }}>
                  <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center", marginBottom:4 }}>
                    <p style={{ color:c.completed?GOLD:textColor, fontSize:12, margin:0, fontFamily:"Georgia,serif", textDecoration:c.completed?"line-through":"none" }}>{c.label}</p>

                </div>
              ))}
            </Card>
          </div>

        {/* ─── TAB 9: Éloges & Méditations ─── */}

                        <p style={{ color:"#8b6914", fontSize:9, letterSpacing:1, margin:"0 0 5px" }}>{p.cat} · {p.ref}</p>
                        <p style={{ color:textColor, fontSize:15, lineHeight:1.8, fontStyle:"italic", margin:"0 0 10px", fontFamily:"'EB Garamond',Georgia,serif" }}>« {p.text} »</p>
                        <div style={{ display:"flex", gap:8 }}>
                          <button onClick={() => { speak(p.text); updateChallenge("check"); }} style={{ background:"transparent", border:"none", color:GOLD, cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>▶ Proclamer</button>
                          <button onClick={() => copyText(`praise-${i}`,p.text)} style={{ background:"transparent", border:"none", color:copiedId===`praise-${i}`?GOLD:"#6b5c3e", cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>
                            {copiedId===`praise-${i}`?"✓":"📋"}
                          </button>
                          <button onClick={() => toggleFav(`praise-${i}`,{text:p.text,label:p.cat})} style={{ background:"transparent", border:"none", color:favorites[`praise-${i}`]?GOLD:"#6b5c3e", cursor:"pointer", fontSize:14 }}>
                            {favorites[`praise-${i}`]?"★":"☆"}
                          </button>
                        </div>
                      </div>
                    </div>
                  </Card>

              </div>
            )}

            {meditTab===1 && (
              <div>
                <Card style={{ marginBottom:10 }}>
                  <Label accent>📖 MÉDITATIONS PROFONDES</Label>

                  </p>
                </Card>
                {MEDITATIONS.map((m,i) => (
                  <Card key={i} style={{ cursor:"pointer", borderColor:expandedMedit===i?GOLD:GOLD_DIM }} onClick={() => setExpandedMedit(expandedMedit===i?null:i)}>
                    <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center" }}>
                      <div>
                        <p style={{ color:GOLD, fontSize:14, fontWeight:"bold", margin:"0 0 2px", fontFamily:"'Cinzel',Georgia,serif" }}>{m.title}</p>
                        <p style={{ color:"#8b6914", fontSize:9, margin:0 }}>{m.ref}</p>
                      </div>
                      <span style={{ color:GOLD, fontSize:14 }}>{expandedMedit===i?"▲":"▼"}</span>
                    </div>
                    {expandedMedit===i && (
                      <div className="fade-in" onClick={e => e.stopPropagation()}>
                        <p style={{ color:textColor, fontSize:14, fontStyle:"italic", lineHeight:1.8, margin:"12px 0 10px", borderLeft:`3px solid ${GOLD_DIM}`, paddingLeft:12, fontFamily:"'EB Garamond',Georgia,serif" }}>
                          « {m.text} »
                        </p>
                        <div style={{ background:"rgba(40,30,10,0.5)", borderRadius:10, padding:"10px 12px", marginBottom:10 }}>
                          <p style={{ color:"#8b6914", fontSize:9, letterSpacing:2, margin:"0 0 6px" }}>RÉFLEXION</p>
                          <p style={{ color:textColor, fontSize:13, lineHeight:1.7, margin:0, fontFamily:"'EB Garamond',Georgia,serif" }}>{m.reflection}</p>
                        </div>
                        <div style={{ display:"flex", gap:8 }}>
                          <button onClick={() => speak(m.text+". "+m.reflection)} style={{ background:"transparent", border:"none", color:GOLD, cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>▶ Méditer</button>

                      </div>
                    )}
                  </Card>
                ))}
              </div>
            )}
          </div>
        )}

        {/* ─── TAB 10: Prières mensuelles ─── */}
        {tab===10 && (
          <div className="fade-in">
            <Card>
              <Label accent>{t.newMonthTitle}</Label>
              <InfoBox title="POURQUOI PRIER POUR LE NOUVEAU MOIS ?" icon="📅">

                  <div style={{ display:"flex", gap:8, marginTop:8 }}>
                    <button onClick={() => speak(p.text)} style={{ background:"transparent", border:"none", color:GOLD, cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>▶ Lire</button>

                  </div>
                </div>
              ))}
            </Card>
            <Card style={{ marginTop:10 }}>
              <Label accent>{t.newYearTitle}</Label>
              <InfoBox title="PROPHÉTISER SA NOUVELLE ANNÉE" icon="🎆">
                La prophétie personnelle est un acte de foi par lequel nous déclarons ce que Dieu a promis pour notre vie. En prophétisant sur votre année, vous activez les promesses divines et positionnez votre vie pour recevoir la plénitude des bénédictions de Dieu.
              </InfoBox>
              <Label>{t.yearlyPrayers}</Label>
              {MONTH_YEAR_PRAYERS.year.map((p,i) => (
                <div key={i} style={{ marginBottom:14 }}>
                  <p style={{ color:GOLD, fontSize:12, margin:"0 0 5px", fontFamily:"'Cinzel',Georgia,serif", letterSpacing:1 }}>{p.title}</p>
                  <p style={{ color:textColor, fontSize:13, lineHeight:1.7, margin:0, fontFamily:"'EB Garamond',Georgia,serif", borderLeft:`2px solid ${GOLD_DIM}`, paddingLeft:10 }}>{p.text}</p>
                  <div style={{ display:"flex", gap:8, marginTop:8 }}>
                    <button onClick={() => speak(p.text)} style={{ background:"transparent", border:"none", color:GOLD, cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>▶ Lire</button>
                    <button onClick={() => copyText(`year-${i}`,p.text)} style={{ background:"transparent", border:"none", color:"#6b5c3e", cursor:"pointer", fontSize:11, fontFamily:"Georgia,serif" }}>📋 Copier</button>
                  </div>
                </div>
              ))}
            </Card>
          </div>
        )}

      </div>

      {/* ── Footer communautaire */}
      <div style={{ position:"fixed", bottom:0, left:0, right:0, background:"rgba(6,3,0,0.98)", borderTop:`1px solid ${GOLD_DIM}`, padding:"10px 14px", backdropFilter:"blur(10px)" }}>
        <p style={{ color:"#6b5c3e", fontSize:8, letterSpacing:2, textAlign:"center", margin:"0 0 7px", fontFamily:"'Cinzel',Georgia,serif" }}>🌐 GROUPES DE PRIÈRE</p>
        <div style={{ display:"flex", gap:8, justifyContent:"center", flexWrap:"wrap" }}>
          <a href="https://t.me/+Y2JoRlZ0DVliNWNk" target="_blank" rel="noopener noreferrer"
            style={{ background:"rgba(0,136,204,0.12)", border:"1px solid rgba(0,136,204,0.3)", borderRadius:18, padding:"5px 14px", color:"#4fc3f7", fontSize:10, textDecoration:"none", fontFamily:"Georgia,serif" }}>
            ✈ Telegram
          </a>
          <a href="https://chat.whatsapp.com/FshwRwqKvYiEBDGptDXoJC?mode=gi_t" target="_blank" rel="noopener noreferrer"
            style={{ background:"rgba(37,211,102,0.1)", border:"1px solid rgba(37,211,102,0.3)", borderRadius:18, padding:"5px 14px", color:"#69f0ae", fontSize:10, textDecoration:"none", fontFamily:"Georgia,serif" }}>
            💬 WhatsApp
          </a>
        </div>
      </div>

      <style>{`
        .fade-in { animation: fadeIn 0.3s ease; }

        select option { background:#1a0f00; color:#e8d4a2; }
      `}</style>
    </div>
  );
